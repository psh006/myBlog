# vueX

# vuex简介

vuex是专门用来管理vue.js应用程序中状态的一个插件，他的作用是将应用中的所有状态都放在一起 集中式来管理,这里所说的状态指的是vue组件中data里面的属性。

# vuex规则

只能在mutaions里修改state,actions不能直接修改state。 mutations即变化,修改state的数据,而且只能是同步,不能存在异步操作,需要异步操作时,将异步操作放在actions里,拿到数据再通过mutations同步处理。

# vuex核心

# 1.1 store

# 1.1.1 创建store

# 1：const store=new Vuex.Store({...})

# 2：store是Vuex.Store这个构造函数new出来的实例,在构造函数中可以传一个对象参数，这个参数可以包含5个对象

# 3：state ---存放状态

# 4：getter ---state的计算属性

# 5：mutations ---更改状态的逻辑,同步操作

# 6：actions ---提交mutation，异步操作

# 7：mudules ---将store模块化

# 注：

store中存储的状态是响应式的，当组件从store中读取状态时，如果store中的状态发生了改变，相应的组件也会得到更新

不能直接改变store中的状态，改变store中状态的唯一途经是提交（commit)mutations,这样使得我们可以方便跟踪每一个状态的变化

# store的完整结构

const store = new Vuex.Store({
  state: {
    // 存放状态
  },
  getters: {
    // state的计算属性
  },
  mutations: {
    // 更改state中状态的逻辑，同步操作
  },
  actions: {
    // 提交mutation，异步操作
  },
  // 如果将store分成一个个的模块的话，则需要用到modules。
   //然后在每一个module中写state, getters, mutations, actions等。
  modules: {
    a: moduleA,
    b: moduleB,
    // ...
  }
});

import {Toast} from 'vant'

// 计算购物车类商品的数量并且重新渲染商品列表
const getShopCarCommodityNumber = (state, productId) => {
  // 计算购物车类商品的数量
  let shopCarCommodityNumber = 0
  // 计算购物车商品的总价格
  let shopCarCommoditySum = 0
  for (const shop of state.shopCar) {
    shopCarCommodityNumber += shop.buyCont
    shopCarCommoditySum += shop.sellPrice * shop.buyCont
  }
  state.shopCarCommodityNumber = shopCarCommodityNumber
  state.shopCarCommoditySum = shopCarCommoditySum
  // 重新渲染商品列表
  let buyCont = 0
  for (const shop of state.shopCar) {
    if (shop.productId === productId) {
      buyCont += shop.buyCont
    }
  }
  for (const shop of state.shopList) {
    if (shop.productId === productId) {
      shop.buyCont = buyCont
    }
  }
}

const state = () => ({
  // 增购或详情对应的商品
  commodityInfo: null,
  // 购物车商品列表
  shopCar: [],
  // 购物车商品列表包含的商品数量
  shopCarCommodityNumber: 0,
  // 购物车商品列表包含的商品的总金额
  shopCarCommoditySum: 0,
  // 商品列表
  shopList: [],
  // 商品标签类型 切换
  productLabel: 0,
  // 商品标签类型 切换 数据
  productLabelItem: {}

})

const mutations = {
  // 设置增购或详情对应的商品
  setCommodityInfo(state, commodity) {
    state.commodityInfo = JSON.parse(JSON.stringify(commodity))
  },
  // 保留 商品标签 选择 状态
  setProductLabel(state, index) {
    state.productLabel = index
  },
  // 保留 选择的标签 数据
  setProductItem(state, productLabelItem) {
    state.productLabelItem = productLabelItem
  },

  // 清空购物车
  clearCommodityInfo(state) {
    state.shopCar = []
    // 购物车商品总数量 置 0
    state.shopCarCommodityNumber = 0
    // 购物车商品合计加购 置0
    state.shopCarCommoditySum = 0
    for (const shop of state.shopList) {
      shop.buyCont = 0
    }
  },
  // 添加商品到购物车
  insertShopCar(state) {
    const commodity = JSON.parse(JSON.stringify(state.commodityInfo))
    for (const shop of state.shopCar) {
      if (shop.productId === commodity.productId && shop.activeProductFlavorName === commodity.activeProductFlavorName) {
        shop.buyCont += 1
        getShopCarCommodityNumber(state, shop.productId)
        return
      }
    }
    state.shopCar.push(commodity)
    getShopCarCommodityNumber(state, commodity.productId)
  },
  // 渲染商品列表
  setShopList(state, shopList) {
    for (const shop of shopList) {
      let buyCont = 0
      for (const item of state.shopCar) {
        if (shop.productId === item.productId) {
          buyCont += item.buyCont
        }
      }
      shop.buyCont = buyCont
    }
    state.shopList = shopList
  },
  // 根据选择商品口味规格,设置该商品购买该商品口味 数量
  setProductFlavorName(state, {index, activeProductFlavorName}) {
    state.commodityInfo.index = index
    state.commodityInfo.activeProductFlavorName = activeProductFlavorName
    for (const shop of state.shopCar) {
      if (shop.productId === state.commodityInfo.productId && shop.activeProductFlavorName === activeProductFlavorName) {
        state.commodityInfo.buyCont = shop.buyCont
        break
      } else {
        state.commodityInfo.buyCont = 0
      }
    }
  },
  // 减少商品
  commodityInfoBuyContReduce(state) {
    const commodity = JSON.parse(JSON.stringify(state.commodityInfo))
    for (const [index, shop] of state.shopCar.entries()) {
      // 商品ID与商品口味都相同，并且当前数量大于0
      if (shop.productId === commodity.productId && shop.activeProductFlavorName === commodity.activeProductFlavorName && shop.buyCont > 0) {
        shop.buyCont -= 1
        state.commodityInfo.buyCont = shop.buyCont
        if (shop.buyCont === 0) {
          state.shopCar.splice(index, 1)
        }
        getShopCarCommodityNumber(state, shop.productId)
      }
    }
  },
  // 增加商品
  commodityInfoBuyContAdd(state, insertShopCar) {
    const buyCont = state.commodityInfo.buyCont
    if (state.commodityInfo.productVariety === 2 || state.commodityInfo.isIgnoreStock === 1) {
      state.commodityInfo.buyCont += 1
      insertShopCar()
    } else if (buyCont > state.commodityInfo.productStock) {
      Toast('商品库存不足!')
    } else {
      state.commodityInfo.buyCont += 1
      insertShopCar()
    }
  }
}

const actions = {
  commodityInfoBuyContAddForShopCar(context) {
    context.commit('commodityInfoBuyContAdd', () => {
      context.commit('insertShopCar')
    })
  }
}

export default {
  namespaced: true,
  state,
  mutations,
  actions
}
