# 正则表达式

# 一、校验数字的表达式

 var num = '125486.6514'
 // 1. 数字：^[0-9]*$
 console.log(num.match(/^[0-9]*$/))
// 2. n位的数字：^\d{n}$
 console.log(num.match(/^\d{8}$/))
 // 3. 至少n位的数字：^\d{n,}$
console.log(num.match(/^\d{3,}$/))
// 4. m-n位的数字：^\d{m,n}$
console.log(num.match(/^\d{4,8}$/))
// 5. 非零开头的数字：^(0|[1-9][0-9]*)$
console.log(num.match(/^([1-9][0-9]*)$/))
// 6. 非零开头的最多带两位小数的数字：^([1-9][0-9]*)+(.[0-9]{1,2})?$
console.log(num.match(/^([1-9][0-9]*)+(.[0-9]{1,2})?$/))
// 7. 带1-2位小数的正数或负数：^(-)?\d+(.\d{1,2})?$
console.log(num.match(/^(-)?\d+(.\d{1,2})?$/))
// 8. 正数、负数、和小数：^(-)?\d+(.\d*)?$
console.log(num.match(/^(-)?\d+(.\d*)?$/))
// 9. 有两位小数的正实数：^[0-9]+(.[0-9]{2})?$
console.log(num.match(/^[0-9]+(.[0-9]{2})?$/))
// 10. 有1~3位小数的正实数：^[0-9]+(.[0-9]{1,3})?$
console.log(num.match(/^[0-9]+(.[0-9]{1,3})?$/))

# 二、校验字符的表达式

var word='dfgg'
// 1. 汉字：^[u4e00-u9fa5]{0,}$
console.log(word.match(/[\u4e00-\u9fa5]+$/))
// 2. 由26个英文字母组成的字符串：^[A-Za-z]+$
console.log(word.match(/^[A-Za-z]+$/))
// 3. 由26个大写英文字母组成的字符串：^[A-Z]+$
console.log(word.match(/^[A-Z]+$/))
// 4. 由26个小写英文字母组成的字符串：^[a-z]+$
console.log(word.match(/^[a-z]+$/))

# 四、强密码

# 1.必须包含大小写字母和数字，特殊字符的组合，长度在8-17之间

function password(word){
//?=.*[a-z]等号后面匹配包含子母的字符串
    var reg=/^(?=.*\d)(?=.*[A-Z])(?=.*[a-z])(?=.*[!@#$%^&*? ]).{8,17}$/;   
		if(reg.test(word)){
		  alert("密码格式正确");
		}else{
		   alert("密码格式不正确");
		}
}
var word='6Q18q66@5485'
console.log(password(word));

# 2.密码应包含大,小写字母,数字,特殊符号($@!%*?&,.`~;:'|)

const isPassword = (number) => {
  const reg = /^[^\s](?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[$@!%*?&,.`~;:'|])[A-Za-z\d$@!%*?&,.`~;:'|]{8,30}$/
  return reg.test(number)
}

const isCardId = (card) => {
  const reg = /^[1-9]\d{7}((0\d)|(1[0-2]))(([0|1|2]\d)|3[0-1])\d{3}$|^[1-9]\d{5}[1-9]\d{3}((0\d)|(1[0-2]))(([0|1|2]\d)|3[0-1])\d{3}([0-9]|X|x)$/

}

# 八、使用@inpute时提供的校验方法

# 1. 校验纯数字

const checkNumber = (value, isString) => {
  const _value = value.replace(/[^\d]/g, '')
  if (isEmpty(_value) || isString) {
    return _value
  }
  return parseInt(_value)
}

# 2. 校验数字+小数点

const checkDecimal = (value) => {
  if (value) {
    value = value.toString()
    value = value.replace(/^\D*(\d*(?:\.\d{0,2})?).*$/g, '$1')
    if (value.length >= 2 && value[0] === '0' && (value[1] === '0' || value[1] !== '.')) {
      return '0'
    }
    return value
  }
}

# 十、js去除字符串空格？

# 方法一：使用replace正则匹配的方法

str = str.replace(/\s*/g,"");  //去除所有空格     
str = str.replace(/^\s*|\s*$/g,""); //去除两头空格
str = str.replace( /^\s*/, ""); //去除左空格
str = str.replace(/(\s*$)/g, ""); //去除右空格

# 方法二：使用str.trim()方法

str.trim()   //局限性：无法去除中间的空格
str.trimLeft(),str.trimRight() //分别用于去除字符串左右空格。
