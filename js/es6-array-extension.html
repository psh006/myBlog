    # es6 数组的扩展

    # Array.from方法将类似数组的对象和可遍历的对象转换为真正的数组（包括NodeList集合、arguments对象等），其实，任何具有length属性的对象都可以转换为数组

    let arrayLike = { '0': 'a', '1': 'b', '2': 'c', length: 3 };
Array.from(arrayLike); // ['a', 'b', 'c']

    # Array.isArray() 用于确定传递的值是否是一个 Array

    Array.isArray([1, 2, 3]);  // true
Array.isArray({foo: 123});  // false

    # Array.of() 返回一个由参数组成的数组

     let arr4 = Array.of(1, 2, 3)  // [1,2,3]

    # find() 用于找出第一个符合条件的数组成员。它的参数是一个回调函数，所有数组成员依次执行该回调函数，直到找出第一个返回值为 true 的成员，然后返回该成员。如果没有符合条件的成员，则返回 undefined。

    [1, 4, -5, 10].find((item)=>{return item <0})  // -5

    # findIndex() 数组实例的 findIndex 方法的用法与 find 方法非常类似，返回第一个符合条件的数组成员的位置，如果所有成员都不符合条件，则返回-1

    [1, 4, -5, 10].findIndex((item)=>{return item < 0 }) // 2

    # copywithin 将数组的某部分替换掉

    let arr4 = Array.of('1', '2', '3', '4');
let arr5 = arr4.copyWithin(3, 1, 3);
//target：开始替换的位置的下标，如题目中的下标3，就是实际上数组中的‘4’；
//start:从哪个位置开始读取数据，如题目中的下标1，就是实际上数组的‘2’

    # fill 将一个数组给初始化为一样的，注意长度

    let arr6 = Array.of('1', '2', '3', '4')
arr6.find((item,index,arr)=>{
if(index == 0) { return false; }
if(index == 1) { return true; }
})
//item:内容 //index:下标 //arr:数组

    # for...of 循环

    // for...of遍历 直接获取到值
var arr8 = Array.of('1', '2', '3', '4');
for(let i of arr8) {console.log(i)} // 1,2,3,4
// for...of搭配keys实现对键的遍历 keys遍历键
for(let i of arr8.keys()) { console.log(i);} // 0,1,2,3
// for...of搭配value实现对值的遍历 遍历值
for(let i of arr8.values()) { console.log(i);} // 1,2,3,4
// for...of搭配entries实现对对象的遍历
for(let i of arr8.entries()) { console.log(i);}
// (2) [0, "1"] // (2) [1, "2"] // (2) [2, "3"] // (2) [3, "4"]

    # JavaScript中常用es6语法整理总结

    # 1,let(用来定义变量)与const(用来定义常量)

    # 相同点:都是块级作用域,在同一个代码块中不允许重复声明
    # 区别
    var //存在变量提升(变量会提升,变量值不会提升),没有块级作用域(可以在 {}号外输出值)，容易造成变量污染
let  //不存在变量提升,定义变量之后才能使用该变量，有作用域（在{}号外面输出会报 is not defined）
const //不存在变量提升，定义变量之后才能使用该变量，定义的常量无法被重新赋值，定义常量时，必须定义且初始化,有块级作用域，如果 const 定义的是一个对象，对象所包含的值是可以被改变的（通俗讲：对象所指向的地址不能改变，里面值可以改变）
