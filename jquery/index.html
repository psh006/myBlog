# 原生JS 和 JQuery的区别

# 1, 入口函数写法格式不同

# 原生JS的固定写法

window.onload=function() {
}

# JQuery的固定写法

$(document).ready(function() {
})
// 简写
$(function() {})

# 2, 获取DOM元素不同

# 原生JS获取元素

var div1 = document.getElementsByClassName('div1')[0];

# JQuery获取元素

var $div1 = $('.div1');

# 3, 修改元素样式不同

# 原生JS修改元素

div1.style.background = 'red'
div1.style.width='300px'

# JQuery修改元素

$div1.css({
		background: 'red',
		width: "300px"
})

# 4, 执行时间不同

原生JS会等DOM元素加载完毕,并且图片也加载完毕才会执行
JQuery会等到Dom元素加载完毕,并不会等到图片也加载完毕就会执行

# 5, 多个入口函数 执行方式

原生的JS如果编写了多个入口函数,后面编写的会覆盖前面编写的
jQuery中编写多个入口函数,后面不会覆盖前面的

# 6, 原生forEach 和 jQuery each用法

var arr = [1, 3, 5, 7, 9];
//伪数组
var obj = {
			0: 1,
			1: 3,
			2: 5,
			3: 7,
			4: 9,
			length: 5
		}

# 原生的forEach方法只能遍历数组,但不能遍历伪数据

arr.forEach((value, index) => {
			console.log(index, value)
		})
arr.forEach((value, index) => {
			console.log(index, value)
		})

# JQuery可以遍历真数组，和伪数组

$.each(arr, function(index, value) {
			console.log('jq arr each', index, value)
		})
$.each(obj, (index, value) => {
			console.log('jq obj each', index, value)
		})
