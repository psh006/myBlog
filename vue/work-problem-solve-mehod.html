# 一, 事件处理

# 1、阻止事件冒泡方法： 在vue中阻止click事件冒泡，防止触发冒泡另一个事件

@click.stop="clickWrapper"

# 2、取消默认事件方法：

@click.prevent="clickWrapper"

# 3、阻止事件冒泡也要取消默认事件

@click.stop.prevent="clickWrapper"

# 四, 批量导出

  exporOut() {
           let idArray = [];
           let contractName = ''
           for (let item of [...this.$refs.listTable.selection]) {
               idArray.push(item.id)
               contractName = item.contractName;
           }
           this.$confirm('确定要导出数据？', {
               confirmButtonText: '确定',
               cancelButtonText: '取消',
           }).then(() => {
               if (this.$util.isEmpty(idArray)) {
                   this.$message.warning("请选择要导出的数据")
               } else {
                   API.expor({idArray: idArray}, (res) => {
                       const aLink = document.createElement("a");
                       let blob = new Blob([res.data], {type: "application/vnd.ms-excel"});
                       aLink.href = URL.createObjectURL(blob);
                       if (this.$refs.listTable.selection.length > 1) {/*多条*/
                           aLink.setAttribute('download', new Date().getTime() + '.xlsx'); // 设置下载文件名称
                       } else {
                           let fileName = contractName;
                           aLink.setAttribute('download', fileName + '.xlsx'); // 设置下载文件名称
                       }
                       aLink.click();
                       this.getContractList();
                   })
               }
           }).catch(() => {
               this.$message.warning('已取消导出数');
           });
       }

                    }
