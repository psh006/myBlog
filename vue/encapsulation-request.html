# 准备一个request.js

    import {Message} from "element-ui"
    import axios from 'axios'
    import util from './util'
    import Router from './../../router/index'
    
    /**
     * 请求API公用函数
     * @param httpMethod 请求方法（GET POST PUT PATCH DELETE） 如果是GET请求并且没有参数 请传空对象{}
     * @param path api路径
     * @param params 请求参数
     * @param success_callback
     * @param fail_callback
     */
    export const requestData = function (params, httpMethod, path, success_callback, fail_callback) {
    
        let key = 'data';
        httpMethod = httpMethod.toUpperCase();
        if (httpMethod === 'GET') {
            key = 'params';
        }
    
        axios({
            url: "/api" + path,
            method: httpMethod,
            withCredentials: true,
            headers: {
                loginToken: util.getUser().loginToken,
            },
            [key]: params
        }).then(res => {
            switch (res.data.code) {
                case 200:
                    success_callback && success_callback(res.data);
                    break;
                case 401:
                case 410:
                case 5002:
                case 5003:
                    Message.warning("登录失效，请重新登录");
                    util.removeUser();
                    Router.push({path: "/login"});
                    break;
                default:
                    Message.error(res.data.message);
                    if (fail_callback) {
                        fail_callback && fail_callback(res.data);
                    }
            }
        }).catch(function (err) {
            console.log(err.response);
            if (err.response) {
                if (err.response.status === 403) {
                    util.removeUser();
                    Router.push({path: "/login"});
                } else if (err.response.status < 500) {
                    Message.error('客户端异常');
                } else {
                    Message.error('服务器异常');
                }
            } else if (err.request) {
                Message.error('服务器未响应');
            } else {
                console.log('------其他错误------');
            }
            if (fail_callback) {
                fail_callback && fail_callback(err);
            }
        });
    };

# 配置代理

在vue.config.js里配置好代理，避免开发模式下的跨域问题

    module.exports = {
        publicPath: process.env.NODE_ENV === 'production' ? '/demo/' : '/',
        outputDir: 'demo',
        devServer: {
            port: 8010,
            disableHostCheck: true,
            proxy: {
                '/api': {
                    //后台接口地址
                    target: "http://127.0.0.1:9000",
                    changeOrigin: true,
                    ws: true,
                    pathRewrite: {
                        '^/api': ''
                    }
                }
            }
        }
    };

# 二次封装

在其他地方二次封装，例如system-setting-api.js,这样可以更好的管理接口


    import {requestData} from "../request";

    //查询用户列表
    const selectAccount = (params, success_callback, error_callback) => {
        requestData(params, "POST", "/accounts/selectAccount", success_callback, error_callback);
    };

    //新增一个用户
    const addAccount = (params, success_callback, error_callback) => {
        requestData(params, "POST", "/accounts", success_callback, error_callback);
    };
    
    export default {
        selectAccount,
        addAccount
    }
