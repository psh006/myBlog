# vue核心配置 vue-cli 3+

# vue.config.js

//当项目部署于ngixn
module.exports = {
  publicPath: process.env.NODE_ENV === '/',
}  

//当项目部署于tomcat
module.exports = {
  publicPath: process.env.NODE_ENV === 'production' ? '/demo/' : '/',
}

//简单完整配置示例(当项目部署于ngixn)
module.exports = {
    publicPath: process.env.NODE_ENV === '/',
    outputDir: 'demo',
    devServer: {
        port: 8080
    }
};

# router/index.js

//当项目部署于ngixn
const router = new VueRouter({
    base: '/',
});

//当项目部署于tomcat
const router = new VueRouter({
    base: '/demo/',
});

//简单完整配置示例(当项目部署于ngixn)
import Vue from 'vue'
import VueRouter from 'vue-router'

Vue.use(VueRouter);

const router = new VueRouter({
    mode: 'history',
    base: '/',
    routes: [
        {
            path: '/home',
            name: 'home',
            component: () => import('../views/home')
        },
        {
            path: '/login',
            name: 'login',
            component: () => import('../views/login')
        },
        {
            path: '/',
            redirect: '/home'
        }
    ]
});

export default router

# 与以上对应的nginx转发配置 443端口

    #当项目部署于ngixn
    location / {
        root /usr/share/nginx/html/demo;
        index index.html;
        try_files $uri $uri/ /index.html;
    }

    #当项目部署于tomcat
    location / {
        #8080是tomcat的端口
        proxy_pass  http://127.0.0.1:8080/;
    }

#简单完整配置示例(当项目部署于ngixn)
server {
  listen 443;
  server_name www.demo.com; #填写绑定证书的域名

  ssl on;
  ssl_certificate ssl_certs/1_www.demo.com_bundle.crt;
  ssl_certificate_key ssl_certs/2_www.demo.com.key;
  ssl_session_timeout 5m;
  ssl_protocols TLSv1 TLSv1.1 TLSv1.2; #按照这个协议配置
  ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:HIGH:!aNULL:!MD5:!RC4:!DHE;#按照这个套件配置
  ssl_prefer_server_ciphers on;


  location / {
    root /usr/share/nginx/html/demo;
    index index.html;
    try_files $uri $uri/ /index.html;
  }
  
  error_page 500 502 503 504  /50x.html;
  location = /50x.html {
	  root html;
  }
}
